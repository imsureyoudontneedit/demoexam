GridView

  • GridView в разы проще ListView, поскольку все данные выводятся в таблицу и необходимо просто запихнуть их в columns dropDownList
<?= GridView::widget([
    'dataProvider' => $dataProvider,
    'filterModel' => $searchModel,
    // Данные для отображжения при отсутствии данных
    'emptyCell' => '',
    // Тоже самое, что и в ListView 
    // В целом большая часть настроек переходит из одного типа вывода в другой
    'layout'=>"{sorter}\n{pager}\n{summary}\n{items}",
    'columns' => [
    ...
        // Простой вариант. Автоматическое формирование изображения
        'category_image:image',
        // Второй вариант. Формирование изображения и его параметров через анонимную функцию
        [
            'label' => 'Картинка',
            'format' => 'raw',
            'value' => function($data){
			    // В целом сложный, но правильный вывод изображения, требует записи полного пути к изображению в бд "@app/web/uploads/img.jpg" 
                return Html::img(Url::toRoute($data->category_image),[
                    'alt'=>'yii2 - картинка в gridview',
                    'style' => 'width:15px;'
                ]);
            },
        ],
    ...
    ],
]); ?>
