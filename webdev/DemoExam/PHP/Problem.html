Создание

  • Через GII создается CRUD-модель и контроллер с ней
  • В форме убираются ненужные поля и добавляем поле с файлом
  • В Zeal пишем Yii:fileUpload и берем все по порядку
  • Также необходимо взять beforeSave он также есть в Zeal
  • Делается функция загрузки изображения в модели, изменяется имя файла на file_name+'.'+file_extension
public $imageFile; 

public function rules() { 
	return [ 
	[['imageFile'], 'file', 'skipOnEmpty' => false, 'extensions' => 'png, jpg'], 
	[['usr_img'], 'string'],
	]; 
}

public function beforeSave($insert) { 
	if (!parent::beforeSave($insert)) { 
		return false; 
	} 
		$this->usr_src = $this->imageFile->name;
		$this->user_id = User::getId();
		// Если необходимо сюда же прописываете статус, но его можно ставить в бд по умолчанию, также как и временную метку
		return true; 
	}

public function upload() {
	if($this->validate()) {
		$fileName = Yii::$app->security->generateRandomString(15).$this->imageFile->extension;
		$this->imageFile->name = $fileName;
		$this->imageFile->saveAs('@app/web/uploads.' . $fileName); 
	}
}
